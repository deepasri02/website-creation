let currentSlide = 0; const slides = document.querySelectorAll('.slide'); function moveSlide(step) { currentSlide += step; if (currentSlide >= slides.length) { currentSlide = 0; } else if (currentSlide < 0) { currentSlide = slides.length - 1; } const sliderContainer = document.querySelector('.slider-container'); sliderContainer.style.transform = `translateX(-${currentSlide * 100}%)`; }
